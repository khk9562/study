const addNewType = (kind) => {
    switch (kind) {
      case "ticket":
        const newTicketId = Math.max(...ticketTypes.map((t) => t.id), 0) + 1;
        setTicketTypes((prevTypes) => [
          ...prevTypes,
          { id: newTicketId, type: `새 타입 ${newTicketId}` },
        ]);
        break;
      case "cabana":
        const newCabanaId = Math.max(...cabanaTypes.map((t) => t.id), 0) + 1;
        setCabanaTypes((prevTypes) => [
          ...prevTypes,
          { id: newCabanaId, type: `새 타입 ${newCabanaId}` },
        ]);
        break;
    }
  };
