1. 접었을 때 CSS 추가하기 위해 jsx로 구현
2. 각 게시글 별로 구현하기 위해 ellipsisList를 데이터 수만큼 false 배열로 넣기
3. 접기, 드러내기 버튼을 눌렀을 때 ellipsisList 아이템들을 true/ false 처리
4. p의 높이별로 버튼 자체를 숨기고 드러내기 위해 showElBtns 배열로
5. useRef로 p 선택하고 높이 측정
6. 높이와 해당 인덱스에 따라 return ref ? ref.offsetHeight > 80 : false;

```
const refs = useRef<(HTMLParagraphElement | null)[]>([]);
  const [ellipsisList, setEllipsisList] = useState<boolean[]>([]);

  const [showElBtns, setShowElBtns] = useState<boolean[]>([]);

  useEffect(() => {
    if (communityData) {
      // Create a new array for the references
      refs.current = communityData.map(
        (_: any, i: number) => refs.current[i] ?? null,
      );

      // Create a new ellipsis list based on the height of each paragraph
      const newShowElBtns = communityData.map((_: any, index: number) => {
        const ref = refs.current[index];
        return ref ? ref.offsetHeight > 80 : false;
      });

      setShowElBtns(newShowElBtns);

      // Initialize ellipsisList as all false
      setEllipsisList(new Array(communityData.length).fill(false));
  // communityData && communityData?.map((_: any, index: number) => false),
    }
  }, [communityData]);

const handleEllipsis = (index: number) => {
    setEllipsisList(prevState =>
      prevState.map((val, i) => (i === index ? !val : val)),
    );
  };
```
